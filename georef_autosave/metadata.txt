name=Georeferencer Autosave

qgisMinimumVersion=3.34
qgisMaximumVersion=4.99
author=Harry Stranger
email=harry@spacefromspace.com
about=Hooks into the QGIS Georeferencer and autosaves GCP points to the standard .points file whenever changes are detected. If QGIS crashes or the Georeferencer is accidentally closed, just reopen and load GCPs as normal. A timestamp can be shown in the Georeferencer status bar so you always know when the last autosave fired.

hasProcessingProvider=no
